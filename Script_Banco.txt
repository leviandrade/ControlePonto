CREATE TABLE TB_COLABORADOR(
	ID_COLABORADOR INT IDENTITY(1,1) NOT NULL,
	NM_COLABORADOR VARCHAR(255) NOT NULL,
	NR_CPF VARCHAR(11) NOT NULL,
	CONSTRAINT PK_TB_COLABORADOR PRIMARY KEY(ID_COLABORADOR)
)

CREATE TABLE TB_REGISTRO_PONTO(
	ID_REGISTRO_PONTO INT IDENTITY(1,1),
	ID_COLABORADOR INT NOT NULL,
	DT_REGISTRO_PONTO DATETIME NOT NULL,
	CONSTRAINT PK_TB_REGISTRO_PONTO PRIMARY KEY(ID_REGISTRO_PONTO),
	CONSTRAINT FK_TB_REGISTRO_PONTO_ID_COLABORADOR FOREIGN KEY(ID_COLABORADOR) REFERENCES TB_COLABORADOR(ID_COLABORADOR)
)
